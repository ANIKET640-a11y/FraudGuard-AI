const fetch = require('node-fetch');

const runRiskAudit = async (amount, time) => {
  const response = await fetch('http://localhost:8000/predict', {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer fg_live_7a3d90f239e083a21bc9e4b',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({ amount, time })
  });
  
  const riskResult = await response.json();
  console.log(`Risk Class: ${riskResult.risk_level} (${riskResult.fraud_probability * 100}%)`);
};

runRiskAudit(1500.00, 36000.0);
import requests

def run_risk_audit(amount: float, time: float):
    url = "http://localhost:8000/predict"
    headers = {
        "Authorization": "Bearer fg_live_7a3d90f239e083a21bc9e4b",
        "Content-Type": "application/json"
    }
    payload = {
        "amount": amount,
        "time": time
    }
    
    res = requests.post(url, headers=headers, json=payload)
    result = res.json()
    print(f"Risk Rating: {result['risk_level']} ({result['fraud_probability'] * 100}%)")

run_risk_audit(1500.00, 36000.0)
